Create Procedure setPersonel
	@pid int,
	@pname nvarchar(20),
	@plname nvarchar(20),
	@p_email nvarchar(40),
	@p_address nvarchar(50),
	@p_dep nvarchar(20),
	@p_task nvarchar(20),
	@p_phone int,
	@p_inphone int


	AS

	SET NOCOUNT ON;
	INSERT INTO dbo.Personel
	(pid,pname,plname,p_email,p_address,p_dep,p_task,p_phone,p_inphone)

	VALUES 
	(@pid,@pname,@plname,@p_email,@p_address,@p_dep,@p_task,@p_phone,@p_inphone)
	SET @pid=@@IDENTITY
	GO
	SET ANSI_NULLS ON
	GO
	SET QUOTED_IDENTIFIER ON
	GO
